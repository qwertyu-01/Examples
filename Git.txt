// инициализация пустого репозитория
git init

// склонировать удаленный репозиторий в текущую директорию
git clone <ссылка на удалённый репозиторий>

// связать удаленный и локальный репозиторий
git remote add origin <ссылка на удалённый репозиторий>

// получить данные из удаленного репозитория
git pull

// получить статус проекта
git status

// добавить в отслеживание
git add . // все файлы
git add <file name 1> <file name 2> //несколько конкретных файлов

// удаление файлов
git rm <filename> <filename>

// фиксация изменений в коммите
git commit -m "Комментарий к коммиту".

// фиксация изменений в коммите без git add . 
// в коммит попадут только изменения уже отслеживаемых файлов
// новые файлы необходимо добавить в отслеживание
git commit -a -m "Комментарий к коммиту".

// отправить изменения на сервер
git push

// забрать ветку из удаленного репозитория
git checkout origin/<branch-name> -b <branch-name>

// переключиться на любую локальную ветку
git checkout <branch-name>

// переключиться на ветку
git switch branch-name

// создать ветку в локальном репозитории
git checkout -b <branch-name>

// отправить ветку в удаленный репозиторий
git push origin <branch-name>

// все ветки локального репозитория
git branch

// все ветки удаленного репозитория
git branch -r

// удалить ветку локального репозитория
git branch -d  <local_branch_name>

// настройка локальной ветки на слежение за удаленной
git branch -u <remote_branch, ex: origin/serverfix>

// посмотреть историю коммитов
git log

// выйти из git log
Q

// сокращенный просмотр коммитов
git log--oneline

// помощь по команде
git <command> -h
